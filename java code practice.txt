class Main {
    public static void main(String[] args) {
        int Arr[] = {1, 4, 7, 8, -4};

        int maxSum = Arr[0];  // Start with the first element
        int currentSum = 0;   // This will hold the current sum we are checking

        for (int i = 0; i < Arr.length; i++) {
            currentSum = currentSum + Arr[i];  // Add current element to currentSum

            if (currentSum > maxSum) {         // If currentSum is bigger than maxSum, update maxSum
                maxSum = currentSum;
            }

            if (currentSum < 0) {               // If currentSum becomes negative, reset it to 0
                currentSum = 0;
            }
        }

        System.out.println("Maximum Subarray Sum: " + maxSum);
    }
}

class Main {
    public static void main(String args[]) {
        int arr[] = {-1, -2, -3, -4};

        int maxSum = arr[0];   // Start with first element
        int currentSum = arr[0];  // Current sum starts with first element

        for (int i = 1; i < arr.length; i++) {
            currentSum = Math.max(arr[i], currentSum + arr[i]);
            maxSum = Math.max(maxSum, currentSum);
        }

        System.out.println("Maximum Subarray Sum: " + maxSum);
    }
}
